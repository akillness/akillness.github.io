~~~py
import torch
import numpy as np
from tqdm import tqdm

# Solve the XOR problem using an MLP with one input/hidden/output layer for classification

X = np.array([[0,0],[1,0],[0,1],[1,1]]) # Input
Y = np.array([[1,0],[0,1],[0,1],[1,0]]) # Output

print('numpy:',X,Y)

X = torch.tensor(X, dtype=torch.float64) # numpy -> torch (double)
Y = torch.tensor(Y, dtype=torch.float64) # numpy -> torch (double)

print('torch:',X,Y)

# MLP
w1 = torch.tensor(np.random.normal(size=(2,2)), requires_grad=True) # Add requires_grad option for training
b1 = torch.tensor(np.random.normal(size=(1,2)), requires_grad=True) # Add requires_grad option for training

print('w1:',w1) #  tensor([[ 0.8241, -1.8039], [ 0.6684, -0.6701]], dtype=torch.float64, requires_grad=True)

w2 = torch.tensor(np.random.normal(size=(2,2))) # Not trainable
b2 = torch.tensor(np.random.normal(size=(1,2))) # Not trainable

print(w2) # tensor([[ 1.2678,  0.4371], [-1.4530, -0.9344]], dtype=torch.float64) # No require_grad

# Add requires_grad for training
w2.requires_grad_(True)

print(w2) # tensor([[ 1.2678,  0.4371], [-1.4530, -0.9344]], dtype=torch.float64, requires_grad=True) # Has requires_grad

# Characteristic of torch functions: if a function ends with '_', it is applied directly to the variable itself. (ex: require_grad, require_grad_)
print(b2) # tensor([[-0.7115], [-0.0490]], dtype=torch.float64)

try:
    b2.requires_grad(True) # The function command exists but using it this way causes an error
except Exception as e:
    print(e)

print('b2:',b2) # tensor([[-0.7115], [-0.0490]], dtype=torch.float64) # Not yet applied
b2.requires_grad_(True)
print('b2:',b2) # tensor([[-0.7115], [-0.0490]], dtype=torch.float64, requires_grad=True) # Applied

print('''[Note]
      Recommended method for neural network weight initialization
      is to use xavier uniform or normal. (See code)
      Remember that the XOR problem also has local minima.
      Could it be overcome by changing hyperparameters?''')

# torch.nn.init.xavier_normal_(w1)
# torch.nn.init.xavier_normal_(b1)
# torch.nn.init.xavier_normal_(w2)
# torch.nn.init.xavier_normal_(b2)

# Preparation for training
opt = torch.optim.Adam([w1,b1,w2,b2]) # Optimizer is Adam; hyperparameters can be set arbitrarily, but using default values for now (learning rate:0.001, beta = (0.9, 0.999), eps=1e-8)

# [Training 1 iteration]
opt.zero_grad()

# Dynamic network computation
xor_mlp_input_layer = X
xor_mlp_hidden_layer = torch.sigmoid(xor_mlp_input_layer@w1+b1)
xor_mlp_output_layer = torch.sigmoid(xor_mlp_hidden_layer@w2+b2)

print('out:',xor_mlp_output_layer) # Current model output

loss = torch.mean((xor_mlp_output_layer-Y)**2) # mean-square loss

print('loss:',loss) # Current model loss
print('loss(value only):',loss.item())

loss.backward()
print('b2(before):',b2)
opt.step()
print('b2(after):',b2)

# [Training 100000 iterations]
for i in (pbar := tqdm(range(100000))):
    opt.zero_grad()

    # Dynamic network computation
    xor_mlp_input_layer = X
    xor_mlp_hidden_layer = torch.sigmoid(xor_mlp_input_layer@w1+b1)
    xor_mlp_output_layer = torch.sigmoid(xor_mlp_hidden_layer@w2+b2)

    loss = torch.mean((xor_mlp_output_layer-Y)**2) # mean-square loss

    loss.backward()
    opt.step()

    if i%1000 == 0:
        pbar.set_description(f'loss:{loss.item()}')

print('Input data:',X)
print('Expected data:',Y)

with torch.no_grad(): # Use no_grad when you don't need training and want to compute without applying gradients
    xor_mlp_input_layer = X
    xor_mlp_hidden_layer = torch.sigmoid(xor_mlp_input_layer@w1+b1)
    xor_mlp_output_layer = torch.sigmoid(xor_mlp_hidden_layer@w2+b2)
    xor_classifier_value = xor_mlp_output_layer

print('XOR neural network result:',xor_classifier_value)
~~~
